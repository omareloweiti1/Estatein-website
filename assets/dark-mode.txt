  <div id="themeToggle" class="flex items-center gap-2 bg-gray-200 dark:bg-gray-700 rounded-full p-1 shadow-inner">
    <button data-theme="system" class="p-2 rounded-full hover:bg-gray-300 dark:hover:bg-gray-600">
      <i class="fa-solid fa-desktop"></i>
    </button>
    <button data-theme="light" class="p-2 rounded-full hover:bg-gray-300 dark:hover:bg-gray-600">
      <i class="fa-solid fa-sun"></i>
    </button>
    <button data-theme="dark" class="p-2 rounded-full hover:bg-gray-300 dark:hover:bg-gray-600">
      <i class="fa-solid fa-moon"></i>
    </button>
  </div>

  <script>
    const page = document.getElementById("page");
    const buttons = document.querySelectorAll("#themeToggle button");

    const prefersDark = window.matchMedia("(prefers-color-scheme: dark)").matches;

    function applyTheme(theme) {
      if (theme === "dark" || (theme === "system" && prefersDark)) {
        page.classList.add("bg-[var(--color-dark-bg)]", "text-white");
        page.classList.remove("bg-[var(--color-light-bg)]", "text-black");
      } else {
        page.classList.add("bg-[var(--color-light-bg)]", "text-black");
        page.classList.remove("bg-[var(--color-dark-bg)]", "text-white");
      }
    }

    applyTheme("system");

    buttons.forEach(btn => {
      btn.addEventListener("click", () => {
        const theme = btn.dataset.theme;
        applyTheme(theme);
      });
    });
  </script>
